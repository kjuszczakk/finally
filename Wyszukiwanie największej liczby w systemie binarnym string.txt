#include <iostream>
#include <fstream>
#include <cstring>
#include <cstdlib>

using namespace std;

int main()
{
    string najw,najm,pom;
	ifstream f1;
	int imax=1,imin=1,i=1;
	f1.open("liczby.txt", ios::in);
	f1>>pom;
	najw=pom;
	najm=pom;
	while (!f1.eof()) {
		f1>>pom;
		i++;
		if (pom.length()>najw.length()) {
				najw=pom;
				imax=i;
		} else if (pom.length()==najw.length()) {
				if (pom>najw) {
						najw=pom;
						imax=i;
				}
		};
		if (pom.length()<najm.length()) {
				najm=pom;
				imin=i;
		} else if (pom.length()==najm.length()) {
				if (pom<najm) {
						najm=pom;
						imin=i;
				}
		};
